    View Problem Statement

    Source: PROGRAMMING.IN.TH

    Difficulty: ?

    Prerequisites:

    • การเขียนโปรแกรมเบื้องต้น
    เฉลย 
    #include <stdio.h>
#include <string.h>

int main() {
    int n;
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        char a[1002];
        scanf("%s", a);
        if (!strcmp(a,"2") || a[strlen(a)-1] % 2 == 1) printf("T\n");
        else printf("F\n");
    }
}

    #include <iostream>

using namespace std;

int main() {
    int n;
    cin >> n;

    for (int i = 0; i < n; i++) {
        string a;
        cin >> a;
        if (a == "2" || a[a.size()-1] % 2 == 1) cout << "T\n";
        else cout << "F\n";
    }
}
