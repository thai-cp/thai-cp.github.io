??? note "เฉลย"
    === "C"
        ```c
        #include <stdio.h>

        int main() {
            int a, b, c;
            scanf("%d %d %d", &a, &b, &c);

            int ans = 0;
            while (a!=1) {ans++; a/=2;}
            while (b!=1) {ans++; b/=2;}
            while (c!=1) {ans++; c/=2;}
            printf("%d", ans);
        }
        ```
    === "C++"
        ```cpp
        #include <iostream>

        using namespace std;

        int main() {
            int a, b, c;
            cin >> a >> b >> c;

            int ans = 0;
            while (a!=1) {ans++; a/=2;}
            while (b!=1) {ans++; b/=2;}
            while (c!=1) {ans++; c/=2;}
            cout << ans;
        }
        ```
