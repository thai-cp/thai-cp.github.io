## เฉลย

??? note "เฉลย"
    === "C"
        ```c
        #include <stdio.h>

        int main() {
            bool rem[42]={};
            int cnt=0;

            for (int i = 0; i < 10; i++) {
                int a;
                scanf("%d", &a);
                if(!rem[a%42]++) cnt++;
            }

            printf("%d", cnt);
        }
        ```
    === "C++"
        ```cpp
        #include <iostream>
        #include <set>

        using namespace std;

        int main() {
            set <int> rem;

            for (int i = 0; i < 10; i++) {
                int a;
                cin >> a;
                rem.insert(a%42);
            }

            cout << rem.size();
        }
        ```
