## เฉลย

??? note "เฉลย"
    เราสามารถรับค่า $n$ หลังจากนั้นวนรับค่า $a$ และเปรียบเทียบ $a$ กับตัวแปร $min$ และ $max$ ทั้งหมด $n$ รอบ
    ??? note "โค้ด"
        === "C"
            ```c
            #include <stdio.h>

            int main() {
                int n;
                scanf("%d", &n);
                int max = −2000000000,min = 2000000000;
                for(int i = 1; i <= n; i++){
                    int a;
                    scanf("%d", &a);
                    if(a > max) max = a;
                    if(a < min) min = a;
                }
                printf("%d\n%d",min,max);
            }
            ```
        === "C++"
            ```cpp
            #include <iostream>

            using namespace std;

            int main(){
                int n;
                cin >> n;
                int max = −2000000000,min = 2000000000;
                for(int i = 1; i <= n; i++){
                    int a;
                    cin >> a;
                    if(a > max) max = a;
                    if(a < min) min = a;
                }
                cout << min << "\n" << max;
            }
            ```
