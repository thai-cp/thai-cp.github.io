## เฉลย

??? note "โค้ดเฉลย"
??? note "เฉลย"
        === "C"
            ```c
            #include <stdio.h>

            int main() {
                int a, b, gcd;
                scanf("%d %d", &a, &b);
                for (int i = 1; i <= a && i <= b; i++) {
                    if (a % i == 0 && b % i == 0) {
                        gcd = i;
                    }
                }
                printf("%d", gcd);
            }
            ```
        === "C++"
            ```cpp
            #include <iostream>

            using namespace std;

            int main() {
                int a, b, gcd;
                cin >> a >> b;
                for (int i = 1; i <= a && i <= b; i++) {
                    if (a % i == 0 && b % i == 0) {
                        gcd = i;
                    }
                }
                cout << gcd;
            }
            ```
