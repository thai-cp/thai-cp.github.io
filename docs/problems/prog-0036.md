??? note "เฉลย"
    === "C"
        ```c
        #include <stdio.h>

        long long nCr(int n, int r) {
            long long res = 1;

            for (int i = n; i > n-r; i--) res *= i;
            for (int i = r; i > 1; i--) res /= i;

            return res;
        }

        int main() {
            int n;
            scanf("%d", &n);

            printf("%lld\n", (n&1 == 1 ? 2*nCr(n, n/2) : nCr(n, n/2)));
        }
        ```
    === "C++"
        ```cpp

        ```
