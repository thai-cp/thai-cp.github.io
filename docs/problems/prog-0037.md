??? note "เฉลย"
    ??? note "โค้ด"
        === "C"
            ```c
            #include <stdio.h>

            int main() {
                int m, n, l, k, c, total = 0;

                scanf("%d %d %d %d %d", &m, &n, &l, &k, &c);

                for (int i = 0; i < m; i++) for (int j = 0; j < n; j++) {
                    int k;
                    scanf("%d", &k);
                    total += k;
                }

                total += l*k*c;

                printf("%d", total/c + (total % c == 0 ? 0 : 1));
            }
            ```
        === "C++"
            ```cpp
            #include <iostream>

            using namespace std;

            int main() {
                int m, n, l, k, c, total = 0;

                cin >> m >> n >> l >> k >> c;

                for (int i = 0; i < m; i++) for (int j = 0; j < n; j++) {
                    int k;
                    cin >> k;
                    total += k;
                }

                total += l*k*c;

                cout << total/c + (total % c == 0 ? 0 : 1);
            }
            ```
