??? note "เฉลย"
    ??? note "โค้ด"
        === "C"
            ```c
            #include <stdio.h>

            int main() {
                int a, b;
                scanf("%d %d", &a, &b);
                printf("%d", b/a + (b%a==0?0:1) + (a>b?1:0));
            }
            ```
        === "C++"
            ```cpp
            #include <iostream>

            using namespace std;

            int main() {
                int a, b;
                cin >> a >> b;
                cout << b/a + (b%a==0?0:1) + (a>b?1:0);
            }
            ```
