??? note "เฉลย"
    === "C"
        ```c
        #include <stdio.h>

        int main() {
            int n;
            scanf("%d", &n);
            int a, ans[10001]={},max=0;

            for (int i = 0; i < n; i++) {
                scanf("%d", &a);
                ans[a]++;
                max=(ans[a]>max)?ans[a]:max;
            }

            for(int i=1;i<=10000;i++)
                if(ans[i]==max) printf("%d ", i);
        }
        ```
    === "C++"
        ```cpp
        #include <iostream>
        #include <vector>

        using namespace std;

        int main() {
            int n;
            cin >> n;
            vector <int> a(n);

            for (int i = 0; i < n; i++) {
                cin >> a[i];
            }

            for(int i=0; i<n-1; i++)
                for(int j=i+1; j<n; j++)
                    if(a[i]>a[j]) swap(a[i],a[j]);

            int m = 0, l = 0, r = 1;
            vector <int> ans;

            while (r <= n) {
                if (r == n || a[l] != a[r]) {
                    if (m < r-l) {
                        ans.clear();
                        ans.push_back(a[l]);
                        m = r-l;
                    } else if (m == r-l) {
                        ans.push_back(a[l]);
                    }
                    l = r;
                }
                r++;
            }

            for (auto i : ans) cout << i << " ";
        }
        ```
