??? note "เฉลย"
    === "C"
        ```c
        #include <stdio.h>

        int main() {
            int n;
            scanf("%d", &n);

            int a[n];
            for (int i = 0; i < n; i++) scanf("%d", &a[i]);

            for(int i=0; i<n-1; i++)
                for(int j=i+1; j<n; j++)
                    if(a[i]>a[j]) {int temp=a[i]; a[i]=a[j], a[j]=temp;}
                    
            int alr=0;

            for(int i=0; i<n; i++){
                if(!alr&&a[i]) printf("%d", a[i]), a[i]=0, i=0, alr=1;
                else if(alr) printf("%d",a[i]);
            }                
        }
        ```
    === "C++"
        ```cpp
        #include <iostream>
        #include <vector>

        using namespace std;

        int main() {
            int n;
            cin >> n;

            vector <int> a(n);
            for (int i = 0; i < n; i++) cin >> a[i];

            for(int i=0;i<a.size()-1;i++)
                for(int j=i+1;j<a.size();j++)
                    if(a[i]>a[j]) swap(a[i],a[j]);

            int alr = 0;

            while (!a.empty()) {
                if (!alr) {
                    for (int i = 0; i < n; i++) {
                        if (a[i] != 0) {
                            cout << a[i];
                            a.erase(a.begin()+i);
                            alr = 1;
                            break;
                        }
                    }
                } else {
                    cout << a[0];
                    a.erase(a.begin());
                }
            }
        }
        ```
