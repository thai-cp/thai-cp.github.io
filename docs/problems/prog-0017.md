        === "C"
            ```c
            #include <stdio.h>

            int main() {
                int side[4];
                for (int i = 0; i < 4; i++)  scanf("%d", &side[i]);
                for(int i=0; i<3; i++)
                    for(int j=i+1; j<4; j++)
                        if(side[i]>side[j]) {int temp=side[i]; side[i]=side[j], side[j]=temp;}

                printf("%d", side[0] * side[2]);
            }
            ```
        === "C++"
            ```cpp
            #include <iostream>

            using namespace std;

            int main() {
                int side[4];
                for (int i = 0; i < 4; i++)  cin >> side[i];
                
                for(int i=0; i<3; i++)
                    for(int j=i+1; j<4; j++)
                        if(side[i]>side[j]) swap(side[i],side[j]);

                cout << side[0] * side[2];
            }
            ```
